\begin{exercise}{1} 
	Set up the linear least squares system $A \bm{x} \cong b$ for fitting the model function  $f(t,\bm{x}) = x_1 t + x_2 e^t$  to the data points (1,2), (2,3), (3,5). \\ 
\end{exercise}

\begin{proof}[]
	\begin{align*}
		A \bm{x} &\cong b \\ 
		\begin{bmatrix}
			1 & e^1 \\
			2 & e^2 \\
			3 & e^3 \\
		\end{bmatrix} 
		\begin{bmatrix}
			x_1 \\
			x_2 \\
		\end{bmatrix} 
		&\cong
		\begin{bmatrix}
			2 \\
			3 \\
			5 \\
		\end{bmatrix}
	\end{align*}
\end{proof}
